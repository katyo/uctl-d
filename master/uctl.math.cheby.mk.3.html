mk

uctl math cheby

Function approximation by fitting Chebyshev polynom in range

  1. template mk(alias C, alias F, real a, real b, uint D)
  2. template mk(alias C, real a, real b, T, alias c)
  3. template mk(alias C, alias F, real a, real b, uint D, T)
    template mk (
    alias C
    alias F
    real a
    real b
    uint D
    T
    ) if (
    __traits(isSame, C, Cheby) &&
    isCallable!F
    &&
    Parameters!F.length == 1
    &&
    isNumer!(Parameters!F[0])
    &&
    isNumer!(ReturnType!F)
    &&
    isNumer!T
    ) {
    enum c;
    }

Examples

Test Chebyshev sin fit (floating-point, 2rd)

1 import std.math: sin;
2 
3 alias xsin = mk!(Cheby, sin, 0, PI*0.5, 2, float);
4 
5 enum pi = float(PI);
6 
7 assert_eq(xsin(0f), -0.016220979f);
8 assert_eq(xsin(pi*1f/6f), 0.512622118f);
9 assert_eq(xsin(pi*1f/4f), 0.707107008f);
10 assert_eq(xsin(pi*1f/3f), 0.854967356f);
11 assert_eq(xsin(pi*1f/2f), 1.010815024f);
12 
13 assert_eq(max_abs_error!(function (float x) => sin(x),
14                          function (float x) => xsin(x))(0.0, PI*0.5, 64),
15           0.016220979f);
16 assert_eq(mean_sqr_error!(function (float x) => sin(x),
17                           function (float x) => xsin(x))(0.0, PI*0.5, 64),
18           9.55991636146791279e-05f);

Test Chebyshev exp2 fit (floating-point, 2nd)

import std.math: exp2;

alias xexp2 = mk!(Cheby, exp2, -1, 1, 2, float);

assert_eq(xexp2(-1f), 0.511991978);
assert_eq(xexp2(1f), 1.983056068);

enum max_err = max_abs_error!(function (float x) => cast(float) exp2(x),
                              function (float x) => xexp2(x))(-1.0, 1.0, 64);
assert_eq(max_err, 0.016944);
enum mean_err = mean_sqr_error!(function (float x) => exp2(x),
                                function (float x) => xexp2(x))(-1.0, 1.0, 64);
assert_eq(mean_err, 0.000101418698020560213);
