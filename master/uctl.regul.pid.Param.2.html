Proportional Integral regulator parameters with integral error limiting

This structure supports builder pattern, so you can construct it and call .with_<property>(<value>) functions in chain. By default all parameter fields is out because it has zero-size. This is important optimization, because it helps save memory. For example, in case of Proportional-Integral (PI) regulator we do not need derivative parameter, so we hide according field from parameters structure by applying none type to it.

  1. struct Param(C, P_)
  2. struct Param(C, P_, I_, E_)
    struct Param (
    C
    P_
    I_
    E_
    ) if (
    isClass!(C, PI) &&
    isNumer!(P_, I_, E_)
    ) {
    alias Class = C;
    alias P = P_;
    alias I = I_;
    alias E = E_;
    P p;
    I i;
    E e;
    this(P p_, I i_, E e_);
    this(P p_, I i_);

    Param!(PID, P, I, D, E) with_D(D d);
    Param!(PID, P, I, typeof(D() * asnum!(1.0 / dt, D)), E) with_D(D d);
    }
