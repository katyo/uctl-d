Examples

Test modulation (floating-point)

1 import uctl: mk;
2 import uctl.util.osc: Param, State;
3 import uctl.math.trig: sin;
4 import uctl.unit: as, qrev, Hz;
5 
6 alias sine = sin!5;
7 enum auto dt = 0.001;
8 
9 auto param = mk!(Param, qrev, dt)(50.0.as!Hz);
10 auto state = State!(param, double)();
11 
12 // Step 0
13 auto phase = state.phase;
14 
15 auto abc = svm!(sine, [3])(phase);
16 assert_eq(abc[0], -0.866197249836520133);
17 assert_eq(abc[1], 0.86619724983652);
18 assert_eq(abc[2], 0.866197249836520133);
19 
20 /// Step 1
21 auto phase2 = state(param);
22 
23 auto abc2 = svm!(sine, [3])(phase2);
24 assert_eq(abc2[0], -0.977943188391134921);
