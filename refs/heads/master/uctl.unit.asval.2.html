Examples

Test asval

alias F = Val!(float, V);
assert(is(typeof(asval!(1.0, F)) == F));
assert(is(typeof(asval!(F, 0.0)) == F));

alias X = Val!(fix!(-1, 1), Ohm);
assert(is(typeof(asval!(1.0, X)) == Val!(fix!1, Ohm)));
assert(is(typeof(asval!(X, 0.0)) == Val!(fix!0, Ohm)));
