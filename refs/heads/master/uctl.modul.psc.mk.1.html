uctl modul psc

Configure corrector using critical time and PWM period

  1. auto mk(T crit_time, D period)
    auto
    mk
    (
    alias P
    T
    D
    )
    (
    const T crit_time
    ,
    const D period
    )
    if (
    __traits(isSame, Param, P) &&
    hasUnits!(T, Time)
    &&
    hasUnits!(D, Time)
    &&
    isNumer!(T.raw_t, D.raw_t)
    )
  2. auto mk(T crit_time, F frequency)
